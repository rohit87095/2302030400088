    <script>
        function checkAnswer() {
            const correctAnswer = "echo";
            const userInput = document.getElementById("userAnswer").value.trim().toLowerCase();
            const responseMessage = document.getElementById("responseMessage");

            if (userInput === correctAnswer) {
                responseMessage.textContent = "Correct! You may proceed to the next challenge.";
                responseMessage.style.color = "green";
            } else {
                responseMessage.textContent = "Wrong answer! Try again.";
                responseMessage.style.color = "red";
            }
        }
    </script>
